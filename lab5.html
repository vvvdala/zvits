int f_recursive(int n)
{
    if (n % 10 > 0) {
        return n % 10;
    }
    else if (n == 0) {
        return 0;
    }
    else {
        return f_recursive(n / 10);
    }
}

int f_iterative(int n)
{
    
    if (n % 10 > 0) {
       return n % 10;
    }
    else if (n == 0) {
       return 0;
    }
    else {
        for (int i = n;;)
        {
            i = i / 10;
            if (i % 10 > 0) {
                 return i %10;
            }
            else if (i == 0) {
               return 0;
            }
        }
    }
}

int body_iterative(int p, int q) {

    int res = 0;
    for (int i = p; i <= q; i++) {
        res += f_iterative(i);
    }
    return res;
}

int body_recursive(int p, int q) {

    int res = 0;
    for (int i = p; i <= q; i++) {
        res += f_recursive(i);
    }
    return res;

    
}
