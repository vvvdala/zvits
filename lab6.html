void shellSort(int N,int arr[]) {
    int j;
   
    for (int step = N / 2;step > 0;step /= 2) {//інтервал
        
        for (int i = step;i < N;i++) {

            int tmp = arr[i];
            for (j = i; j >= step; j -= step)
            {
                if (tmp < arr[j - step]) {//якщо елемент менше з порівнювальним, то міняємо місцями
                    arr[j] = arr[j - step];
                }
                else {
                    break;
                }
                
            }
            arr[j] = tmp;
            
               
           
        }
        for (int i = 0; i < N; i++)
        {

            cout << arr[i] << "\t";

        }
        cout << "\n";
    }

}

int jumpSearch(int arr[], int N, int item) {
    int i = 0;
    int step = sqrt(N);

    while (arr[step] <= item && step < N) {
        //цикл для продовження "пригання" по блоках
        i = step;
        step += sqrt(N);


    }

    for (int x = i;x < step;x++) {//лінійний пошук в даному блоці
        if (arr[x] == item) {
            return x;
        }
       
    }
     
    
}

int summa_f(int N, int arr[],int x) {

    float summa=0;
    
    for (int i = 0; i < N; i++) {

        summa += arr[i]*pow(x,(N-i));
         

    }
     

    return summa;
}

int summa_g(int N, int arr[],int x) {
   

    float summa = 0;
    

    for (int i = 0; i < N; i++) {

        summa += arr[i] * pow(x, (N - i));

    }


    return summa;
    
}

int summa_h(int N, int arr[],int x) {

    float summa = 0;
   
    for (int i = 0; i < N; i++) {

        summa += arr[i] * pow(x, (N - i));

    }




    return summa;
}


int vlast1(int fx[], int gx[], int N, int N1,int x) {
    int left = summa_f(N, fx,x)+ summa_g(N1, gx,x);
    int right = summa_g(N1, gx,x)+ summa_f(N, fx,x);
    
    cout << "Left side:";
    cout << left << endl;

    cout << "Right side:";
    cout << right << endl;

   

    if (left == right) {
        cout << "right!" << endl;

    }
    else {
        cout << "wrong";
    }

    return 0;

}

int vlast2(int fx[], int gx[], int hx[], int N, int N1, int N3,int x) {
    int left = summa_f(N, fx,x);
    int left1 = summa_g(N1, gx,x);
    int left2 = summa_h(N3, hx,x);

    int right = summa_g(N1, gx,x);
    int right1 = summa_f(N, fx,x);
    int right2 = summa_h(N3, hx,x);

    cout << "Left side:";
    cout << left * (left1 * left2) << endl;

    cout << "Right side:";
    cout << (right * right1) * right2 << endl;


    cout << "So our property is...";

    if (left * (left1 * left2) == (right * right1) * right2) {
        cout << "right!" << endl;

    }
    else {
        cout << "wrong";
    }
    return 0;

}


int vlast3(int fx[], int gx[], int hx[], int N, int N1, int N3,int x) {
    int left = summa_f(N, fx,x);
    int left1 = summa_g(N1, gx,x);
    int left2 = summa_h(N3, hx,x);

    int right = summa_f(N, fx,x);
    int right1 = summa_g(N1, gx,x);
    int right2 = summa_h(N3, hx,x);

    cout << "Left side:";
    cout << left * (left1 + left2) << endl;

    cout << "Right side:";
    cout << (right * right1) + (right * right2) << endl;


    cout << "So our property is...";

    if (left * (left1 + left2) == (right * right1) + (right * right2)) {
        cout << "right!" << endl;

    }
    else {
        cout << "wrong";
    }
    return 0;

}
